00.(vision, research, game selection, architecture){
    We have now completed Phase 0 — Vision, Research, System Design, and Architecture Finalization for our offline multi-game AI platform project.
    The project direction evolved from a simple “play games against computer AI” idea into a fully structured, modular, Windows-compatible strategy game ecosystem focused entirely on deterministic 1v1 board-combat games where human players compete against AI agents in pure skill-based environments.
    After deep filtering and analysis, we finalized a carefully curated set of nine games that match the exact psychological and gameplay feel we wanted: Chess, Tic-Tac-Toe, Connect Four, Checkers, Othello, Gomoku, Pente, Go, and Nine Men's Morris.
    we rejected puzzle-style and randomness-heavy games because they broke the “board war against an opponent” philosophy.we
    we also decided the project must remain platform-independent and self-contained for Windows users, leading to the selection of Python + Tkinter + PyInstaller as the core technology stack.
    Architecturally, we finalized a professional modular design separating engine logic, game logic, and UI systems into independent layers, ensuring scalability, maintainability, reusable AI systems, and future extensibility.
    Shared board abstractions (grid_board, graph_board, intersection_board) were introduced to avoid duplication across different game types, while universal AI interfaces and game interfaces were planned to support plug-and-play intelligence systems such as Minimax, heuristic AI, and future advanced algorithms.
    we also restructured the roadmap into a cleaner 15-phase development lifecycle where this finalized planning and architectural work officially becomes the new Phase 0, establishing the project foundation before implementation begins.

        01(create time.py){
            "|"
            import time
            class TimeManager:
                def __init__(self):
                    self.delta_time = 0.0
                    self.current_time = time.perf_counter()
                    self.previous_time = self.current_time
                    self.fps = 0
                    self.frame_count = 0
                    self.fps_timer = 0.0
                    self.total_runtime = 0.0
                def update(self):
                    self.current_time = time.perf_counter()
                    self.delta_time = (self.current_time - self.previous_time)
                    self.previous_time = self.current_time
                    self.total_runtime += self.delta_time
                    self.frame_count += 1
                    self.fps_timer += self.delta_time
                    if self.fps_timer >= 1.0:
                        self.fps = self.frame_count
                        self.frame_count = 0
                        self.fps_timer = 0.0
                def get_delta_time(self):
                    return self.delta_time
                def get_fps(self):
                    return self.fps
                def get_runtime(self):
                    return self.total_runtime
            "|"

            import pygame
            class Input:
                _current_keys = None
                _previous_keys = None
                _current_mouse_buttons = (False, False, False)
                _previous_mouse_buttons = (False, False, False)
                _mouse_position = (0, 0)
                @classmethod
                def initialize(cls):
                    cls._current_keys = pygame.key.get_pressed()
                    cls._previous_keys = pygame.key.get_pressed()
                    cls._current_mouse_buttons = pygame.mouse.get_pressed()
                    cls._previous_mouse_buttons = pygame.mouse.get_pressed()
                    cls._mouse_position = pygame.mouse.get_pos()
                    print("[Input] Initialized")
                @classmethod
                def update(cls):
                    cls._previous_keys = cls._current_keys
                    cls._current_keys = pygame.key.get_pressed()
                    cls._previous_mouse_buttons = cls._current_mouse_buttons
                    cls._current_mouse_buttons = pygame.mouse.get_pressed()
                    cls._mouse_position = pygame.mouse.get_pos()
                @classmethod
                def is_key_pressed(cls, key):
                    return (cls._current_keys[key]andnot cls._previous_keys[key])
                @classmethod
                def is_key_held(cls, key):
                    return cls._current_keys[key]
                @classmethod
                def is_key_released(cls, key):
                    return (not cls._current_keys[key]andcls._previous_keys[key])
                @classmethod
                def is_mouse_pressed(cls, button):
                    index = button - 1
                    return (cls._current_mouse_buttons[index] and not cls._previous_mouse_buttons[index])
                @classmethod
                def is_mouse_held(cls, button):
                    index = button - 1
                    return cls._current_mouse_buttons[index]
                @classmethod
                def is_mouse_released(cls, button):
                    index = button - 1
                    return (not cls._current_mouse_buttons[index] and cls._previous_mouse_buttons[index])
                @classmethod
                def get_mouse_position(cls):
                    return cls._mouse_position
            "|"

            import pygame

                Logger.info(

            def load_image(
                cls,
                asset_name,
                relative_path
            ):
                if asset_name in cls._images:
                    Logger.warning(
                        f"[AssetManager] Image already loaded: {asset_name}"
                    )
                    return
                path = Path(relative_path)
                if not path.exists():
                    Logger.error(
                        f"[AssetManager] Missing image: {relative_path}"
                    )

                cls._images[asset_name] = pygame.image.load(
                    path
                )
                Logger.info(
                    f"[AssetManager] Loaded image: {asset_name}"
                )

            def get_image(
                cls,
                asset_name

                return cls._images.get(
                    asset_name
                )

                cls,
                asset_name,
                relative_path,
                size
            ):
                key = f"{asset_name}_{size}"
                if key in cls._fonts:
                    Logger.warning(
                        f"[AssetManager] Font already loaded: {key}"
                    )
                    return

                    )
                    crash_file = crash_dir / "crash.log"

    Phase 1 of MindWar Arena focused on building a stable, scalable, and reusable foundation for the entire project. 
    The goal was not to create a playable game yet, but to establish the core engine architecture that every future game, AI system, board representation, UI component, and gameplay feature will rely upon. 
    By the end of Phase 1, the project evolved from an empty folder structure into a functioning game engine capable of initializing, managing resources, handling input, running scenes, rendering graphics through OpenGL, managing time, logging events, handling crashes, and supporting future game and AI integrations through standardized interfaces.

    The first major task completed during this phase was the project environment setup. 
    A dedicated Python virtual environment was created to isolate project dependencies and ensure consistent development across systems. 
    The latest versions of pip, setuptools, and wheel were installed and updated to provide a stable package management environment. 
    Core libraries required for the future engine were installed, including pygame for window creation and input handling, moderngl for OpenGL rendering, numpy for mathematical operations, pillow for image processing, pyglm for vector and matrix mathematics, colorama for terminal output enhancements, and rich for structured console output. 
    Once installation was complete, a dedicated environment validation script was created and executed successfully, confirming that all libraries imported correctly and that the development environment was fully operational.

    After the environment was verified, the project architecture was established. 
    The repository was organized into clearly separated modules, including engine, game, games, ui, assets, config, logs, saves, tests, docs, and build directories. 
    This structure was designed with long-term scalability in mind. 
    The engine folder became the heart of the framework, containing reusable systems that are independent of any specific game implementation. 
    Future board games such as Chess, Checkers, Go, Gomoku, Pente, Othello, Connect Four, Tic Tac Toe, and Nine Men’s Morris were allocated their own isolated directories, ensuring that game-specific logic remains separated from engine functionality.

    The next milestone involved implementing the engine boot system. 
    This system serves as the entry point of the application and is responsible for initializing all engine components in the correct order. 
    The launcher module was created as the startup coordinator, while the GameManager became the central controller responsible for engine initialization, runtime execution, and shutdown procedures. 
    This established the fundamental lifecycle of the application and created a single location from which all future systems can be controlled.

    Window creation and rendering functionality were then added. 
    A dedicated Window class was implemented using pygame. 
    This system manages window creation, screen dimensions, display settings, frame timing, and shutdown operations. 
    OpenGL support was configured through pygame’s OpenGL context settings, ensuring compatibility with modern rendering techniques. 
    Alongside the window system, a Renderer class was created using moderngl. 
    The renderer initializes an OpenGL context, reports hardware-supported OpenGL information, and provides the ability to clear the display each frame. 
    This established the graphical foundation required for future rendering systems, board visualization, animations, and user interface components.

    With rendering operational, a main game loop was implemented. 
    The game loop became the continuous execution cycle responsible for processing events, updating systems, rendering frames, and enforcing frame-rate limits. 
    The loop follows the traditional architecture used in professional game engines and serves as the heartbeat of the application. 
    Through this implementation, the engine gained the ability to run continuously until a shutdown condition occurs, providing a framework upon which all future gameplay systems will operate.

    Input management was then introduced through the Input system. 
    This module tracks keyboard and mouse states, supports detection of key presses, releases, and holds, and provides access to mouse position and mouse button states. 
    The system separates current and previous input states, allowing precise event detection. 
    During testing, keyboard input including W, A, S, D, and Escape keys was successfully detected, along with mouse click events and cursor position tracking. 
    This established a robust input foundation suitable for menus, gameplay interactions, camera controls, and future UI systems.

    Scene management was implemented next. 
    A generic Scene base class was created to define the lifecycle of scenes through enter, update, render, and exit operations. 
    A SceneManager was introduced to control scene transitions and maintain the currently active scene. 
    This architecture allows the engine to switch between different states such as boot screens, main menus, gameplay screens, settings menus, pause screens, and future multiplayer lobbies. 
    To validate the system, a BootScene was created and successfully integrated into the engine startup sequence.

    Configuration management followed shortly after. 
    A dedicated Config system was implemented to load settings from JSON configuration files located within the config directory. 
    The settings.json file became the primary source of engine configuration values such as window title, screen dimensions, and frame-rate limits. 
    The Config class provides centralized access to configuration values, ensuring that engine settings can be modified without changing source code. 
    This system lays the groundwork for future graphics settings, audio options, gameplay preferences, AI configurations, and user customization features.

    Logging infrastructure was another major milestone. 
    A dedicated Logger utility was created to standardize information output across the entire project. 
    The logger supports multiple severity levels including debug, info, warning, error, and critical. 
    All major engine systems were integrated with the logging framework, replacing direct print statements with structured log entries. 
    Engine activity is recorded both to the console and to persistent log files, creating a reliable debugging and monitoring mechanism. 
    This system greatly improves maintainability and provides valuable diagnostics during development and future testing.

    Resource management capabilities were introduced through the AssetManager. 
    This module centralizes loading and retrieval of images and fonts while preventing duplicate asset loading. 
    The manager maintains internal caches and tracks loaded resources, allowing future systems to access shared assets efficiently. 
    Error handling was included to report missing resources and prevent crashes due to invalid asset references. 
    Although no production assets were loaded during Phase 1, the infrastructure required for future content management is now fully established.

    To support future AI and game implementations, a standardized interface architecture was created. 
    The GameInterface defines the requirements that all future game modules must satisfy, including initialization, resetting, updating, rendering, state retrieval, game-over detection, and winner determination. 
    Similarly, the AIInterface establishes a common structure for all AI systems, including initialization, action selection, learning behavior, and reset functionality. 
    These interfaces provide a consistent contract that future games and AI agents must follow, ensuring compatibility across the entire platform.

    During development, temporary validation classes named TestGame and TestAI were implemented to verify interface compliance. 
    These classes successfully demonstrated that the interface architecture functioned as intended. Once validation was complete, the temporary testing classes were removed as part of the Phase 1 cleanup process, leaving only the permanent interface definitions.

    Time management functionality was implemented through the TimeManager. 
    This system tracks delta time, frame counts, frames per second, and total application runtime. 
    By calculating frame timing independently of rendering speed, the engine is prepared for frame-rate-independent gameplay logic. 
    The system also updates the window title with live FPS information, providing immediate performance feedback during development.

    Crash handling and exception management were added to improve engine stability. 
    A dedicated CrashHandler utility was created to capture unhandled exceptions, generate crash reports, and store diagnostic information within the logs directory. 
    Rather than allowing unexpected exceptions to terminate the application without explanation, the crash handler records timestamps, exception details, and stack traces. 
    Testing confirmed that intentional exceptions generated proper crash reports and were handled gracefully. 
    This system provides an important safety net for future development and debugging.

    A comprehensive integration testing stage was conducted before finalizing the foundation. 
    The boot sequence was temporarily transformed into a validation framework that tested configuration loading, asset management, interface compliance, engine startup, and system integration. 
    Successful execution confirmed that all major subsystems were communicating correctly. 
    Once validation was completed, the testing framework was removed from BootScene to restore its intended role as a startup scene.

    A cleanup and stabilization pass concluded the phase. 
    Temporary testing files were deleted, startup validation logic was removed, and project structure was reviewed for consistency. 
    Logging files were preserved but cleared, maintaining future diagnostic capability without retaining unnecessary historical data. 
    A version file was added to track engine metadata, including engine name, version number, project phase, and completion status. 
    Scene naming conventions were standardized by renaming the main menu scene file to align with the scene architecture. 
    Empty but strategically important folders were preserved to support future phases of development.

    By the completion of Phase 1, MindWar Arena possessed a fully operational engine foundation featuring environment management, dependency control, project architecture, application bootstrapping, OpenGL rendering, window management, event processing, input handling, scene management, configuration systems, asset management, logging, crash recovery, timing systems, interface contracts, and engine lifecycle management. 
    The engine can initialize successfully, create a rendering context, enter a scene, process user input, maintain frame timing, log activity, and shut down cleanly.

    Most importantly, Phase 1 established the architectural principles that will guide the remainder of the project. 
    Systems are modular, responsibilities are clearly separated, interfaces define contracts between components, utilities are centralized, and future expansion paths are already prepared. 
    This foundation ensures that upcoming phases can focus on implementing gameplay mechanics, board systems, AI algorithms, user interfaces, game rules, and advanced features without needing to revisit core engine design.

    Phase 1 can therefore be considered a complete success. 
    The project now stands on a stable, maintainable, and scalable technical foundation, ready to begin Phase 2, where the focus will shift from engine infrastructure toward the creation of the core gameplay framework that will eventually support multiple board games and AI competitors within the MindWar Arena ecosystem.

            class TicTacToeRules:
                @staticmethod
                def check_winner(board):
                    for row in range(BOARD_ROWS):
                        first = board.get_cell(row, 0)
                        if first == EMPTY:
                            continue
                        winner = True
                        for column in range(1, BOARD_COLUMNS):
                            if board.get_cell(row, column) != first:
                                winner = False
                                break
                        if winner:
                            return first
                    for column in range(BOARD_COLUMNS):
                        first = board.get_cell(0, column)
                            if first == EMPTY:
                            continue
                        winner = True
                        for row in range(1, BOARD_ROWS):
                            if board.get_cell(row, column) != first:
                                winner = False
                                break
                        if winner:
                            return first
                    first = board.get_cell(0, 0)
                    if first != EMPTY:
                        winner = True
                        for index in range(1, BOARD_ROWS):
                            if board.get_cell(index, index) != first:
                                winner = False
                                break
                        if winner:
                            return first
                    first = board.get_cell(0, BOARD_COLUMNS - 1)
                    if first != EMPTY:
                        winner = True
                        for index in range(1, BOARD_ROWS):
                            if board.get_cell(index, BOARD_COLUMNS - 1 - index) != first:
                                winner = False
                                break
                        if winner:
                            return first
                    return NO_WINNER
                @staticmethod
                def is_draw(board):
                    return (
                        TicTacToeRules.check_winner(board) == NO_WINNER

                        board.is_board_full()

                def is_game_over(board):
                    if TicTacToeRules.check_winner(board) != NO_WINNER:

                                name: str,
                                vertex_shader_path: str,
                                fragment_shader_path: str,
                            ):
                                if name in self._programs:
                                    Logger.warning(
                                        f"[ShaderManager] Shader '{name}' already loaded."
                                    )
                                    return self._programs[name]
                                vertex_source = self._read_shader(vertex_shader_path)
                                fragment_source = self._read_shader(fragment_shader_path)
                                program = self.context.program(
                                    vertex_shader=vertex_source,
                                    fragment_shader=fragment_source,
                                )
                                self._programs[name] = program
                                Logger.info(
                                    f"[ShaderManager] Loaded shader '{name}'."
                                )
                                return program
                            def get_program(self, name: str):
                                return self._programs.get(name)
                            def has_program(self, name: str):
                                return name in self._programs
                            def release(self):
                                for name, program in self._programs.items():
                                    try:
                                        program.release()
                                        Logger.info(
                                            f"[ShaderManager] Released '{name}'."
                                        )
                                    except Exception as exception:
                                        Logger.warning(
                                            f"[ShaderManager] Failed to release '{name}': {exception}"
                                        )
                                self._programs.clear()

                        from __future__ import annotations
                        import numpy as np
                        from engine.utils.logger import Logger
                        class Mesh:
                            def __init__(
                                self,
                                context,
                                program,
                                vertices,
                                vertex_format: str = "2f",
                                attributes=("in_position",),
                                dynamic: bool = True,
                            ):
                                self.context = context
                                self.program = program
                                self.vertex_format = vertex_format
                                self.attributes = attributes
                                self.dynamic = dynamic
                                self.vbo = None
                                self.vao = None
                                self.buffer_size = 0
                                self._create(vertices)
                            def _create(self, vertices):
                                data = np.array(vertices, dtype="f4")
                                raw_data = data.tobytes()
                                self.buffer_size = len(raw_data)
                                self.vbo = self.context.buffer(
                                    raw_data,
                                    dynamic=self.dynamic
                                )
                                self.vao = self.context.vertex_array(
                                    self.program,
                                    [
                                        (
                                            self.vbo,
                                            self.vertex_format,
                                            *self.attributes,
                                        )
                                    ],
                                )
                                Logger.info("[Mesh] GPU mesh created.")
                            def update_vertices(self, vertices):
                                data = np.array(vertices, dtype="f4")
                                raw_data = data.tobytes()
                                if len(raw_data) > self.buffer_size:
                                    self.buffer_size = len(raw_data)
                                    if self.vao is not None:
                                        self.vao.release()
                                    if self.vbo is not None:
                                        self.vbo.release()
                                    self.vbo = self.context.buffer(
                                        raw_data,
                                        dynamic=self.dynamic,
                                    )
                                    self.vao = self.context.vertex_array(
                                        self.program,
                                        [
                                            (
                                                self.vbo,
                                                self.vertex_format,
                                                *self.attributes,
                                            )
                                        ],
                                    )
                                    Logger.debug(
                                        f"[Mesh] GPU buffer resized to {self.buffer_size} bytes."
                                    )
                                else:
                                    self.vbo.write(raw_data)
                            def render(self, mode):
                                self.vao.render(mode=mode)
                            def release(self):
                                try:
                                    if self.vao is not None:
                                        self.vao.release()
                                    if self.vbo is not None:
                                        self.vbo.release()
                                    Logger.info("[Mesh] Released.")
                                except Exception as exception:
                                    Logger.warning(
                                        f"[Mesh] Release failed: {exception}"
                                    )

                from games.tic_tac_toe.constants import (
                    BOARD_ROWS,
                    BOARD_COLUMNS,
                    CELL_SIZE,
                    BOARD_PADDING,
                    GRID_COLOR,
                    X_COLOR,
                    O_COLOR,
                    EMPTY,
                    PLAYER_X,
                    PLAYER_O,
                )
                class BoardRenderer:
                    def __init__(self, renderer):
                        self.renderer = renderer
                        self.origin_x = BOARD_PADDING
                        self.origin_y = BOARD_PADDING
                    def render(self, board):
                        self.draw_grid()
                        for row in range(BOARD_ROWS):
                            for column in range(BOARD_COLUMNS):
                                value = board.get_cell(row, column)
                                if value == PLAYER_X:
                                    self.draw_x(row, column)
                                elif value == PLAYER_O:
                                    self.draw_o(row, column)
                    def draw_grid(self):
                        self.renderer.draw_grid(
                            origin=(self.origin_x, self.origin_y),
                            rows=BOARD_ROWS,
                            columns=BOARD_COLUMNS,
                            cell_size=CELL_SIZE,
                            color=GRID_COLOR,
                        )
                    def draw_x(self, row, column):
                        center = self.get_cell_center(row, column)
                        self.renderer.draw_cross(
                            center=center,
                            size=CELL_SIZE * 0.60,
                            color=X_COLOR,
                        )
                    def draw_o(self, row, column):
                        center = self.get_cell_center(row, column)
                        self.renderer.draw_circle(
                            center=center,
                            radius=CELL_SIZE * 0.30,
                            color=O_COLOR,
                        )
                    def get_cell_center(self, row, column):
                        x = (
                            self.origin_x
                            + column * CELL_SIZE
                            + CELL_SIZE / 2
                        )
                        y = (
                            self.origin_y
                            + row * CELL_SIZE
                            + CELL_SIZE / 2
                        )
                        return (x, y)
                    def contains_point(self, x, y):
                        board_width = BOARD_COLUMNS * CELL_SIZE
                        board_height = BOARD_ROWS * CELL_SIZE
                        return (
                            self.origin_x <= x < self.origin_x + board_width
                            and
                            self.origin_y <= y < self.origin_y + board_height
                        )
                    def screen_to_cell(self, x, y):
                        if not self.contains_point(x, y):
                            return None
                        column = int((x - self.origin_x) // CELL_SIZE)
                        row = int((y - self.origin_y) // CELL_SIZE)
                        return (row, column)

                    BOARD_ROWS,

                class TicTacToeRules:
                    @staticmethod
                    def check_winner(board):
                        for row in range(BOARD_ROWS):
                            first = board.get_cell(row, 0)
                            if first == EMPTY:
                                continue
                            winner = True
                            for column in range(1, BOARD_COLUMNS):
                                if board.get_cell(row, column) != first:
                                    winner = False
                                    break
                            if winner:
                                return first
                        for column in range(BOARD_COLUMNS):
                            first = board.get_cell(0, column)
                            if first == EMPTY:
                                continue
                            winner = True
                            for row in range(1, BOARD_ROWS):
                                if board.get_cell(row, column) != first:
                                winner = False
                            break
                            if winner:
                                return first
                        first = board.get_cell(0, 0)
                        if first != EMPTY:
                            winner = True
                            for index in range(1, BOARD_ROWS):
                                if board.get_cell(index, index) != first:
                                    winner = False
                                    break
                            if winner:
                                return first
                        first = board.get_cell(0, BOARD_COLUMNS - 1)
                        if first != EMPTY:
                            winner = True
                            for index in range(1, BOARD_ROWS):
                                if board.get_cell(index, BOARD_COLUMNS - 1 - index) != first:
                                    winner = False
                                    break
                            if winner:
                                return first
                        return NO_WINNER

                    def is_draw(board):
                        return (
                            TicTacToeRules.check_winner(board) == NO_WINNER
                            and
                            board.is_board_full()

                    def is_game_over(board):

                    def initialize(cls):

                from __future__ import annotations
                import numpy as np
                from engine.utils.logger import Logger
                class Mesh:
                    def __init__(
                        self,
                        context,
                        program,
                        vertices,
                        vertex_format: str = "2f",
                        attributes=("in_position",),
                        dynamic: bool = True,
                    ):
                        self.context = context
                        self.program = program
                        self.vertex_format = vertex_format
                        self.attributes = attributes
                        self.dynamic = dynamic
                        self.vbo = None
                        self.vao = None
                        self.buffer_size = 0
                        self._create(vertices)
                    def _create(self, vertices):
                        data = np.array(vertices, dtype="f4")
                        raw_data = data.tobytes()
                        self.buffer_size = len(raw_data)
                        self.vbo = self.context.buffer(
                            raw_data,
                            dynamic=self.dynamic
                        )
                        self._create_vertex_array()
                        Logger.info("[Mesh] GPU mesh created.")
                    def _create_vertex_array(self):
                        self.vao = self.context.vertex_array(
                            self.program,
                            [
                                (
                                    self.vbo,
                                    self.vertex_format,
                                    *self.attributes,
                                )
                            ],
                        )
                        return self.vao
                    def update_vertices(self, vertices):
                        data = np.array(vertices, dtype="f4")
                        raw_data = data.tobytes()
                        if len(raw_data) > self.buffer_size:
                            self.buffer_size = len(raw_data)
                            if self.vao is not None:
                                self.vao.release()
                            if self.vbo is not None:
                                self.vbo.release()
                            self.vbo = self.context.buffer(
                                raw_data,
                                dynamic=self.dynamic,
                            )
                            self._create_vertex_array()
                            Logger.debug(
                                f"[Mesh] GPU buffer resized to {self.buffer_size} bytes."
                            )
                        else:
                            self.vbo.write(raw_data)
                    def rebuild(self, program):
                        self.program = program
                        if self.vao is not None:
                            self.vao.release()
                        self._create_vertex_array()
                    @property
                    def vertex_count(self):
                        stride = sum(
                            int(part[:-1])
                            for part in self.vertex_format.split()
                        )
                        return self.buffer_size // (4 * stride)
                    def render(self, mode):
                        self.vao.render(mode=mode)
                    def release(self):
                        try:
                            if self.vao is not None:
                                self.vao.release()
                                self.vao = None
                            if self.vbo is not None:
                                self.vbo.release()
                                self.vbo = None
                            Logger.info("[Mesh] Released.")
                        except Exception as exception:
                            Logger.warning(
                                f"[Mesh] Release failed: {exception}"
                            )
                "|"

                    self,

                        color=GRID_COLOR,

                        color=X_COLOR,

                        color=O_COLOR,
                    )

                    x = (
                        self.origin_x
                        + column * CELL_SIZE
                        + CELL_SIZE / 2
                    )
                    y = (
                        self.origin_y
                        + row * CELL_SIZE
                        + CELL_SIZE / 2
                    )
                    return (x, y)

                    board_height = BOARD_ROWS * CELL_SIZE

            from __future__ import annotations
            import numpy as np
            from engine.utils.logger import Logger
            class Mesh:
                def __init__(self,context,program,vertices,vertex_format: str = "2f",attributes=("in_position",),dynamic: bool = True, ):
                    self.context = context
                    self.program = program
                    self.vertex_format = vertex_format
                    self.attributes = attributes
                    self.dynamic = dynamic
                    self.vbo = None
                    self.vao = None
                    self.buffer_size = 0
                    self.vertex_count = 0
                    self._create(vertices)
                def _create(self, vertices):
                    data = np.array(vertices, dtype="f4")
                    raw_data = data.tobytes()
                    self.buffer_size = len(raw_data)
                    self.vbo = self.context.buffer(raw_data,dynamic=self.dynamic)
                    self._create_vertex_array()
                    Logger.info("[Mesh] GPU mesh created.")
                def _create_vertex_array(self):
                    self.vao = self.context.vertex_array(self.program,[(self.vbo,self.vertex_format,*self.attributes, )], )
                    return self.vao
                def update_vertices(self, vertices):
                    data = np.array(vertices, dtype="f4")
                    raw_data = data.tobytes()
                    self.vertex_count = len(vertices) // 2
                    if len(raw_data) > self.buffer_size:
                        self.buffer_size = len(raw_data)
                        if self.vao is not None:
                            self.vao.release()
                        if self.vbo is not None:
                            self.vbo.release()
                        self.vbo = self.context.buffer(raw_data,dynamic=self.dynamic, )
                        self._create_vertex_array()
                        Logger.debug(f"[Mesh] GPU buffer resized to {self.buffer_size} bytes.")
                    else:
                        self.vbo.write(raw_data)
                def rebuild(self, program):
                    self.program = program
                    if self.vao is not None:
                        self.vao.release()
                    self._create_vertex_array()
                def render(self, mode):
                    self.vao.render(mode=mode,vertices=self.vertex_count, )
                def release(self):
                    try:
                        if self.vao is not None:
                            self.vao.release()
                            self.vao = None
                        if self.vbo is not None:
                            self.vbo.release()
                            self.vbo = None
                        Logger.info("[Mesh] Released.")
                    except Exception as exception:
                        Logger.warning(f"[Mesh] Release failed: {exception}")
