00.(vision, research, game selection, architecture){
    We have now completed Phase 0 — Vision, Research, System Design, and Architecture Finalization for our offline multi-game AI platform project.
    The project direction evolved from a simple “play games against computer AI” idea into a fully structured, modular, Windows-compatible strategy game ecosystem focused entirely on deterministic 1v1 board-combat games where human players compete against AI agents in pure skill-based environments.
    After deep filtering and analysis, we finalized a carefully curated set of nine games that match the exact psychological and gameplay feel we wanted: Chess, Tic-Tac-Toe, Connect Four, Checkers, Othello, Gomoku, Pente, Go, and Nine Men's Morris.
    we rejected puzzle-style and randomness-heavy games because they broke the “board war against an opponent” philosophy.we
    we also decided the project must remain platform-independent and self-contained for Windows users, leading to the selection of Python + Tkinter + PyInstaller as the core technology stack.
    Architecturally, we finalized a professional modular design separating engine logic, game logic, and UI systems into independent layers, ensuring scalability, maintainability, reusable AI systems, and future extensibility.
    Shared board abstractions (grid_board, graph_board, intersection_board) were introduced to avoid duplication across different game types, while universal AI interfaces and game interfaces were planned to support plug-and-play intelligence systems such as Minimax, heuristic AI, and future advanced algorithms.
    we also restructured the roadmap into a cleaner 15-phase development lifecycle where this finalized planning and architectural work officially becomes the new Phase 0, establishing the project foundation before implementation begins.

                    pygame.display.flip()
                def tick(self, fps=60):
                    self.clock.tick(fps)

                    pygame.quit()

        01(create time.py){
            "|"
            import time
            class TimeManager:
                def __init__(self):
                    self.delta_time = 0.0
                    self.current_time = time.perf_counter()
                    self.previous_time = self.current_time
                    self.fps = 0
                    self.frame_count = 0
                    self.fps_timer = 0.0
                    self.total_runtime = 0.0
                def update(self):
                    self.current_time = time.perf_counter()
                    self.delta_time = (
                        self.current_time - self.previous_time
                    )
                    self.previous_time = self.current_time
                    self.total_runtime += self.delta_time
                    self.frame_count += 1
                    self.fps_timer += self.delta_time
                    if self.fps_timer >= 1.0:
                        self.fps = self.frame_count
                        self.frame_count = 0
                        self.fps_timer = 0.0
                def get_delta_time(self):
                    return self.delta_time
                def get_fps(self):
                    return self.fps
                def get_runtime(self):
                    return self.total_runtime
            "|"
        }

            "|"
            import pygame
            class Input:
                _current_keys = None
                _previous_keys = None
                _current_mouse_buttons = (False, False, False)
                _previous_mouse_buttons = (False, False, False)
                _mouse_position = (0, 0)
                @classmethod
                def initialize(cls):
                    cls._current_keys = pygame.key.get_pressed()
                    cls._previous_keys = pygame.key.get_pressed()
                    cls._current_mouse_buttons = pygame.mouse.get_pressed()
                    cls._previous_mouse_buttons = pygame.mouse.get_pressed()
                    cls._mouse_position = pygame.mouse.get_pos()
                    print("[Input] Initialized")
                @classmethod
                def update(cls):
                    cls._previous_keys = cls._current_keys
                    cls._current_keys = pygame.key.get_pressed()
                    cls._previous_mouse_buttons = cls._current_mouse_buttons
                    cls._current_mouse_buttons = pygame.mouse.get_pressed()
                    cls._mouse_position = pygame.mouse.get_pos()
                @classmethod
                def is_key_pressed(cls, key):
                    return (
                        cls._current_keys[key]
                        and
                        not cls._previous_keys[key]
                    )
                @classmethod
                def is_key_held(cls, key):
                    return cls._current_keys[key]
                @classmethod
                def is_key_released(cls, key):
                    return (
                        not cls._current_keys[key]
                        and
                        cls._previous_keys[key]
                    )
                @classmethod
                def is_mouse_pressed(cls, button):
                    index = button - 1
                    return (
                        cls._current_mouse_buttons[index]
                        and
                        not cls._previous_mouse_buttons[index]
                    )
                @classmethod
                def is_mouse_held(cls, button):
                    index = button - 1
                    return cls._current_mouse_buttons[index]
                @classmethod
                def is_mouse_released(cls, button):
                    index = button - 1
                    return (
                        not cls._current_mouse_buttons[index]
                        and
                        cls._previous_mouse_buttons[index]
                    )
                @classmethod
                def get_mouse_position(cls):
                    return cls._mouse_position
            "|"
        }

            import pygame
            from engine.utils.logger import Logger
            class Input:
                _current_keys = None
                _previous_keys = None
                _current_mouse_buttons = (False, False, False)
                _previous_mouse_buttons = (False, False, False)
                _mouse_position = (0, 0)
                @classmethod
                def initialize(cls):
                    cls._current_keys = pygame.key.get_pressed()
                    cls._previous_keys = pygame.key.get_pressed()
                    cls._current_mouse_buttons = pygame.mouse.get_pressed()
                    cls._previous_mouse_buttons = pygame.mouse.get_pressed()
                    cls._mouse_position = pygame.mouse.get_pos()
                    Logger.info("[Input] Initialized")
                @classmethod
                def update(cls):
                    cls._previous_keys = cls._current_keys
                    cls._current_keys = pygame.key.get_pressed()
                    cls._previous_mouse_buttons = cls._current_mouse_buttons
                    cls._current_mouse_buttons = pygame.mouse.get_pressed()
                    cls._mouse_position = pygame.mouse.get_pos()
                @classmethod
                def is_key_pressed(cls, key):
                    return (
                        cls._current_keys[key]
                        and
                        not cls._previous_keys[key]
                    )
                @classmethod
                def is_key_held(cls, key):
                    return cls._current_keys[key]
                @classmethod
                def is_key_released(cls, key):
                    return (
                        not cls._current_keys[key]
                        and
                        cls._previous_keys[key]
                    )
                @classmethod
                def is_mouse_pressed(cls, button):
                    index = button - 1
                    return (
                        cls._current_mouse_buttons[index]
                        and
                        not cls._previous_mouse_buttons[index]
                    )
                @classmethod
                def is_mouse_held(cls, button):
                    index = button - 1
                    return cls._current_mouse_buttons[index]
                @classmethod
                def is_mouse_released(cls, button):
                    index = button - 1
                    return (
                        not cls._current_mouse_buttons[index]
                        and
                        cls._previous_mouse_buttons[index]
                    )
                @classmethod
                def get_mouse_position(cls):
                    return cls._mouse_position
